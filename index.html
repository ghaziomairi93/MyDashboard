el.innerHTML+=`<div class="reminder-item" onclick="toggleR(${idx})"><div class="check ${r.done?'done':''}">${r.done?'✓':''}</div><div style="flex:1"><div class="reminder-title ${r.done?'done':''}">${r.title}</div><div class="reminder-due ${r.cls==='overdue'?'due-overdue':r.cls==='today'?'due-today':'due-upcoming'}">${r.icon} ${r.due}</div></div><div>${r.p}</div></div>`;
});
}
function toggleR(i){reminders[i].done=!reminders[i].done;reminders[i].cat=reminders[i].done?‘done’:‘today’;reminders[i].due=reminders[i].done?‘Completed’:‘Today’;reminders[i].cls=reminders[i].done?‘upcoming’:‘today’;renderReminders(rFilter);}
function filterR(f,btn){rFilter=f;document.querySelectorAll(’.chip’).forEach(b=>b.classList.remove(‘active’));btn.classList.add(‘active’);renderReminders(f);}
renderReminders(‘all’);
